Copy the component source into your project:

```bash

```

## Usage

```tsx
import { Orb } from "@/components/ui/orb"

export default function App() {
  return (
    <div style={{ width: 300, height: 300 }}>
      <Orb />
    </div>
  )
}
```

## Examples

### Custom Colors

Specify gradient colors via the `colors` prop.

<OrbColorsDemo client:load />

```tsx
<Orb colors={["#FF6B6B", "#4ECDC4"]} />
```

### Agent State

Manage visual states to reflect the agent's current activity.

<OrbAgentStateDemo client:load />

```tsx
import { useState } from "react"
import { Orb, type AgentState } from "@/components/ui/orb"

export function AgentOrb() {
  const [state, setState] = useState<AgentState>(null)

  return (
    <div>
      <div style={{ width: 300, height: 300 }}>
        <Orb agentState={state} />
      </div>
      <div>
        <button onClick={() => setState(null)}>Idle</button>
        <button onClick={() => setState("thinking")}>Thinking</button>
        <button onClick={() => setState("listening")}>Listening</button>
        <button onClick={() => setState("talking")}>Talking</button>
      </div>
    </div>
  )
}
```

### Manual Volume Control

Directly control input and output volume values with sliders.

<OrbVolumeDemo client:load />

```tsx
<Orb
  volumeMode="manual"
  manualInput={0.5}
  manualOutput={0.8}
/>
```

### Custom Seed

Use a fixed seed for consistent animation patterns across renders.

```tsx
<Orb seed={42} />
```

### Audio Reactivity

Use callback functions to provide real-time volume data.

```tsx
<Orb
  volumeMode="auto"
  getInputVolume={() => microphoneLevel}
  getOutputVolume={() => speakerLevel}
/>
```

### AgentState

```ts
type AgentState = null | "thinking" | "listening" | "talking"
```

| State | Description |
|-------|-------------|
| `null` | Idle state with subtle ambient animation. |
| `"thinking"` | Gentle wandering motion indicating processing. |
| `"listening"` | Responsive animation simulating active listening. |
| `"talking"` | Dynamic animation simulating speech output. |

## Notes

- The component uses raw **WebGL** shaders â€” no Three.js dependency required.
- A Perlin noise texture is loaded from an external CDN for the flow distortion effect.
- The canvas automatically handles **device pixel ratio** for sharp rendering on high-DPI displays.
- Colors transition smoothly via linear interpolation when updated.
- The component properly cleans up WebGL resources on unmount.
